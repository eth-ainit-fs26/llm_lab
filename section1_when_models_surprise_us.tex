% --- SLIDE 7: THE CoT DISCOVERY -----------------------------------------------
\begin{frame}{Chain-of-Thought Prompting}

\begin{columns}[T]
  \begin{column}{0.48\textwidth}
    \textbf{Standard prompting:}\\[0.3em]
    {\small\texttt{Q: Roger has 5 tennis balls.\\
    He buys 2 cans of 3 balls.\\
    How many does he have?}\\[0.3em]
    \texttt{A: 11}}

    \vspace{0.8em}
    \textbf{Chain-of-thought:}\\[0.3em]
    {\small\texttt{A: Roger started with 5.\\
    2 cans $\times$ 3 = 6 new balls.\\
    5 + 6 = \alert{11}. The answer is 11.}}
  \end{column}
  \begin{column}{0.48\textwidth}
    \begin{tabular}{lc}
      \toprule
      \textbf{Method} & \textbf{GSM8K} \\
      \midrule
      Standard & ${\sim}$18\% \\
      Chain-of-Thought & ${\sim}$\alert{58\%} \\
      Prior SOTA & 55\% \\
      \bottomrule
    \end{tabular}

    \vspace{0.8em}
    {\small Zero fine-tuning. Just 8 hand-written
    exemplars with reasoning steps.}

    \vspace{0.5em}
    {\scriptsize \papercite{Wei et al.}{2022}{NeurIPS}{Chain-of-Thought Prompting Elicits Reasoning in LLMs}}
  \end{column}
\end{columns}

\end{frame}

% --- SLIDE 13: VERIFICATION METHODS ------------------------------------------
\begin{frame}{How Do You Verify Reasoning?}

\begin{columns}[T]
  \begin{column}{0.48\textwidth}
    \textbf{Outcome Reward Models (ORMs)}\\[0.3em]
    {\small Score the \textbf{final answer} only.\\
    Binary: correct or incorrect.\\
    Simple but coarse.}

    \vspace{1em}
    \textbf{Process Reward Models (PRMs)}\\[0.3em]
    {\small Score \textbf{each reasoning step}.\\
    Catches errors early. 78\% on MATH.}

    \vspace{0.3em}
    {\scriptsize \papercite{Lightman et al.}{2023}{OpenAI}{Let's Verify Step by Step}}
  \end{column}
  \begin{column}{0.48\textwidth}
    \textbf{Generative PRMs}\\[0.3em]
    {\small Use a long-CoT model to \emph{verify}
    each step via its own reasoning.
    Strong verification, far fewer labels.}

    \vspace{0.8em}
    \textbf{Why it matters}\\[0.3em]
    {\small Process supervision outperforms outcome
    supervision. Scoring each step is key to
    scaling test-time compute.}
  \end{column}
\end{columns}

\end{frame}

% --- SLIDE 15: TWO AXES OF SCALING -------------------------------------------
\begin{frame}{Two Axes of Scaling}

{\small The capability frontier now has \textbf{two dimensions}:}

\vspace{0.3em}

\begin{columns}[T]
  \begin{column}{0.48\textwidth}
    \textbf{Train-time compute}\\[0.2em]
    {\small One-time capital expense.\\
    Bigger models: GPT-2 $\to$ GPT-4.\\
    More parameters, data, FLOPs.}
  \end{column}
  \begin{column}{0.48\textwidth}
    \textbf{Test-time compute}\\[0.2em]
    {\small Per-query operating expense.\\
    Think longer: o1, DeepSeek-R1.\\
    Same model, more reasoning tokens.}
  \end{column}
\end{columns}

\vspace{0.5em}

{\small \textbf{Strategic implications:}}
\begin{itemize}\setlength{\itemsep}{0.1em}
  \item {\small \textbf{High-volume, low-latency}: invest in a fast, large model.}
  \item {\small \textbf{High-stakes, low-volume}: smaller model + extended thinking.}
  \item {\small Test-time compute is most effective for \alert{medium-difficulty} problems.}
\end{itemize}

\end{frame}

  \item[\textbf{1.}] If we cannot predict what the next model will be capable
    of, \alert{how should organisations plan for AI adoption}?

  \vspace{0.8em}
  \item[\textbf{2.}] Chain-of-thought evolved from a prompting trick to a
    training paradigm in under 3 years.
    \alert{What reasoning capabilities are next}?

  \vspace{0.8em}
  \item[\textbf{3.}] Test-time compute trades latency for accuracy.
    \alert{Which of your business decisions warrant a model that ``thinks
    longer''}?
\end{enumerate}

\end{frame}
