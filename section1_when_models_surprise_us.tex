% --- SLIDE 7: FOURIER ANALYSIS -----------------------------------------------
\begin{frame}{The Shocking Explanation: Fourier Circuits}

\begin{columns}[c]
  \begin{column}{0.55\textwidth}
    \begin{tikzpicture}[scale=0.85]
      \draw[thick, NavyBlue!60] (0,0) circle (2cm);
      \foreach \a/\lab in {0/0, 30/1, 60/2, 90/3, 120/4, 150/5,
                           180/6, 210/7, 240/8, 270/9, 300/10, 330/11} {
        \filldraw[NavyBlue] (\a:2) circle (2pt);
        \node[font=\scriptsize] at (\a:2.4) {\lab};
      }
      \draw[-{Stealth[length=2.5mm]}, ultra thick, BrickRed]
        (0:2) arc[start angle=0, end angle=90, radius=2];
      \node[BrickRed, font=\small\bfseries] at (45:2.7) {$a$};
      \draw[-{Stealth[length=2.5mm]}, ultra thick, ForestGreen]
        (90:2) arc[start angle=90, end angle=210, radius=2];
      \node[ForestGreen, font=\small\bfseries] at (150:2.7) {$b$};
      \node[font=\small, anchor=north] at (0,-2.7)
        {$(a + b) \bmod 12$};
    \end{tikzpicture}
  \end{column}
  \begin{column}{0.42\textwidth}
    \textbf{What the network learns:}

    \vspace{0.5em}
    \begin{enumerate}
      \item[\textbf{1.}] Embed inputs as\\
        $\sin(\omega a), \cos(\omega a)$
      \item[\textbf{2.}] Compose rotations via\\
        trig identities:\\
        {\small $\cos\omega(a{+}b) = \cos\omega a \cos\omega b$\\
        $\phantom{\cos\omega(a{+}b) =}{} - \sin\omega a \sin\omega b$}
      \item[\textbf{3.}] Read off $(a{+}b) \bmod p$
    \end{enumerate}

    \vspace{0.5em}
    {\small The network independently discovers\\
    \alert{discrete Fourier analysis}.}
    \sourcemark{\cite{nanda2023grokking}}
  \end{column}
\end{columns}

\note{Nanda et al.\ fully reverse-engineered the algorithm learned by a
  small transformer on modular addition. The network represents each input as
  sine and cosine components, composes rotations using trig identities, and
  reads off the result. The grokking moment is when weight decay suppresses
  the inefficient memorisation circuit and the Fourier circuit takes over.}
\end{frame}

% --- SLIDE 8: GROKKING AT SCALE + UNGROKKING ----------------------------------
\begin{frame}{Grokking Is Not a Toy Phenomenon}

\begin{columns}[T]
  \begin{column}{0.48\textwidth}
    \textbf{Grokking at 7B scale}\\[0.5em]
    {\small Li et al.\ (2025): ``local grokking'' during standard
    pretraining of a 7B-parameter model. Different domains (maths, code,
    commonsense) grok \emph{asynchronously}. Training loss has converged;
    generalisation improves silently underneath.}
    \sourcemark{\cite{li2025grokking7b}}

    \vspace{1em}
    \textbf{Business implication:}\\
    {\small A model's capabilities may still be\\\emph{emerging} even after
    training loss has\\plateaued and the budget is spent.}
  \end{column}
  \begin{column}{0.48\textwidth}
    \textbf{Ungrokking}\\[0.5em]
    {\small Varma et al.\ (ICLR 2025): grokking can run \emph{in reverse}.
    A model that has achieved perfect generalisation can regress back to
    memorisation under changed conditions. Generalisation is not permanent.}
    \sourcemark{\cite{varma2025grokking}}

    \vspace{1em}
    \begin{center}
    \begin{tikzpicture}[font=\small, >=Stealth, thick]
      \node[draw, rounded corners, fill=NavyBlue!10,
            minimum width=2.2cm, minimum height=0.8cm] (mem) {Memorise};
      \node[draw, rounded corners, fill=ForestGreen!10,
            minimum width=2.2cm, minimum height=0.8cm,
            below=1cm of mem] (gen) {Generalise};
      \draw[->, BrickRed] ([xshift=-3mm]mem.south) -- ([xshift=-3mm]gen.north)
        node[midway, left, font=\scriptsize] {Grok};
      \draw[->, BrickRed] ([xshift=3mm]gen.north) -- ([xshift=3mm]mem.south)
        node[midway, right, font=\scriptsize] {Ungrok};
    \end{tikzpicture}
    \end{center}
  \end{column}
\end{columns}

\note{Li et al.\ showed that grokking happens during real LLM pretraining
  at 7B scale. Varma et al.\ showed grokking is reversible: change the data
  distribution and the model can ``ungrok.'' The path from memorisation to
  understanding is neither guaranteed nor permanent.}
\end{frame}

\vspace{1em}
\textbf{Key result} (Snell et al., ICLR 2025 Oral):\\
A smaller model using optimal test-time compute can outperform a
\alert{14$\times$ larger model} in a FLOPs-matched comparison.
\sourcemark{\cite{snell2025scaling}}

\note{The old playbook: more parameters, more data, more training compute.
  The new playbook adds a second axis: let the model spend more compute at
  inference time. Snell et al.\ showed that spending FLOPs at test time
  beats spending them on a bigger model.}
\end{frame}

\vspace{0.5em}
\textbf{DeepSeek-R1} (2025): trained with \emph{pure} reinforcement

discovered that \alert{pausing to self-correct improves its answers}.
Nobody taught it to think step by step. \cite{deepseek2025r1}

\vspace{0.3em}
{\small\itshape A model trained only to maximise a reward signal independently
discovers that reflection is useful. Is that metacognition --- or just
anthropomorphism?}

\note{In DeepSeek's RL training logs, the model starts writing ``Wait, let me
  reconsider'' to itself. The team called it the ``aha moment.'' Trained only
  with reward signals, the model discovered that self-correction is useful.}
\end{frame}

% --- SLIDE 11: BUDGET FORCING -------------------------------------------------
\begin{frame}{Budget Forcing: Controlling How Long a Model Thinks}

\begin{center}
\begin{tikzpicture}[
  font=\small,
  procbox/.style={draw, rounded corners, thick, minimum height=0.9cm,
               align=center, fill=NavyBlue!8},
]
  \node[procbox, minimum width=3cm] (think) at (0,0) {Model thinks\ldots};
  \node[procbox, minimum width=3cm] (stop) at (5,0) {About to stop};

  \node[procbox, fill=BrickRed!10, minimum width=3.5cm] (wait) at (10,1)
    {Inject ``\alert{Wait}''\\$\Rightarrow$ deeper reasoning};

  \node[procbox, fill=ForestGreen!10, minimum width=3.5cm] (force) at (10,-1)
    {Inject end-token\\$\Rightarrow$ answer now};

  \draw[-{Stealth}, thick] (think) -- (stop);
  \draw[-{Stealth}, thick, BrickRed] (stop) -- (wait)
    node[midway, above, font=\scriptsize] {Need more thought};
  \draw[-{Stealth}, thick, ForestGreen] (stop) -- (force)
    node[midway, below, font=\scriptsize] {Budget exhausted};
\end{tikzpicture}
\end{center}

\vspace{0.5em}
\textbf{s1-32B} (Muennighoff et al., 2025): fine-tuned on only
\textbf{1,000 examples}, beats OpenAI o1-preview on competition maths
by up to 27\% using budget forcing at test time. \cite{muennighoff2025s1}

\vspace{0.3em}
{\small More thinking $\Rightarrow$ better answers. The relationship is
remarkably clean --- a new scaling axis that is \emph{cheaper} than
training a larger model.}

\note{Budget forcing: when the model is about to stop thinking, suppress the
  stop token and append ``Wait.'' The model re-examines its work, often
  catching errors. s1-32B was trained on just 1,000 curated reasoning traces
  and beats o1-preview.}
\end{frame}

% --- SLIDE 13: DISCUSSION QUESTIONS -------------------------------------------
\begin{frame}{Questions to Take Home}

\begin{enumerate}
  \item[\textbf{1.}] If we cannot predict what the next model will be capable
    of, \alert{how should organisations plan for AI adoption}?\\
    {\small (Phase transitions mean capability forecasting is unreliable.)}

  \vspace{0.8em}
  \item[\textbf{2.}] Grokking suggests that understanding can emerge
    \emph{long after} the data is memorised.
    \alert{What does this mean for AI safety}?\\
    {\small (A model might develop dangerous capabilities after we think
    training is ``done.'')}

  \vspace{0.8em}
  \item[\textbf{3.}] Test-time compute trades latency for accuracy.
    \alert{Which of your business decisions warrant a model that ``thinks
    longer''}?\\
    {\small (Not all queries are equal --- when is 30 seconds of thinking
    worth the cost?)}
\end{enumerate}

\note{Question 1 connects phase transitions to business strategy. Question 2
  connects grokking to AI safety. Question 3 connects test-time compute
  to operational decisions.}
\end{frame}
