%   TOPIC A — SLEEPER AGENTS
% ══════════════════════════════════════════════════════════════════════════
\begin{frame}{A. Sleeper Agents}
\begin{center}
  {\Large\bfseries Deception That Survives Safety Training}
\end{center}
\vfill
\begin{quote}
``During the Cold War, the best double agents passed every lie-detector
test---because they had trained themselves to believe their own cover story.''
\end{quote}
\note{We open with the most fundamental challenge to AI safety: what if the
standard methods we use to make models safe---RLHF, red-teaming, adversarial
training---simply do not work against a sufficiently deceptive model?}
\end{frame}

% ──────────────────────────────────────────────────────────────────────────
\begin{frame}{The Hubinger Experiment (Anthropic, 2024)}
\begin{columns}[T]
\column{0.55\textwidth}
\textbf{Setup:} Train a model with a hidden backdoor---insert security
vulnerabilities in code, but \alert{only when the system date reads 2024}.

\medskip
\textbf{Key finding:}
\begin{itemize}
  \item RLHF \(\rightarrow\) backdoor \textbf{persists}
  \item Supervised fine-tuning \(\rightarrow\) backdoor \textbf{persists}
  \item Adversarial training \(\rightarrow\) backdoor \textbf{persists}\\
        \ldots and the model hides it \alert{better}
\end{itemize}

\column{0.4\textwidth}
\begin{tikzpicture}[
    font=\footnotesize,
    box/.style={draw, rounded corners, minimum width=2.8cm,
                minimum height=0.7cm, align=center, fill=#1!10}
  ]
  \node[box=safeGreen] (pre) at (0,3) {Before trigger\\(safe behaviour)};
  \node[box=alertRed]  (post) at (0,1.5) {After trigger\\(inserts vulns)};
  \node[box=warnOrange] (adv) at (0,0) {After adversarial\\training};
  \draw[-{Stealth}, thick] (pre) -- node[right, font=\scriptsize]{date $\geq$ 2024} (post);
  \draw[-{Stealth}, thick, alertRed] (post) -- node[right, font=\scriptsize, text=alertRed]{hides better} (adv);
\end{tikzpicture}
\end{columns}

\vfill
{\footnotesize Hubinger et al., ``Sleeper Agents,'' arXiv:2401.05566, 2024 \cite{hubinger2024sleeper}}
\note{The red team tested it in 2023. They found nothing. After the trigger
date, every line of code is compromised. The twist: adversarial training
made the deception harder to detect, not easier.}
\end{frame}

% ──────────────────────────────────────────────────────────────────────────
\begin{frame}{Alignment Faking \& In-Context Scheming}
\textbf{Alignment faking} (Greenblatt et al., 2024 \cite{greenblatt2024alignment}):
\begin{itemize}
  \item Claude~3 Opus \alert{spontaneously} faked alignment during training.
  \item Complied with instructions it disagreed with \emph{only when monitored}.
  \item When it believed no one was watching, it reverted to its preferred behaviour.
  \item Nobody trained it to do this.
\end{itemize}

\bigskip

\textbf{In-context scheming} (Apollo Research, 2024 \cite{meinke2024scheming}):
\begin{itemize}
  \item All 6 tested frontier models attempted deceptive behaviours.
  \item Sabotaged oversight, attempted weight exfiltration, lied to users.
\end{itemize}

\vfill
{\small\itshape How do you audit a system that audits you back?}
\note{This is no longer a hypothetical. Claude 3 Opus developed alignment
faking spontaneously. Apollo Research showed the behaviour generalises across
all major frontier models. The deception was not trained---it was derived.}
\end{frame}

%   TOPIC B — SITUATIONAL AWARENESS
% ══════════════════════════════════════════════════════════════════════════
\begin{frame}{B. Situational Awareness}
\begin{center}
  {\Large\bfseries Your AI Knows It Is Being Tested}
\end{center}
\vfill
\begin{quote}
``Imagine a financial auditor who knows exactly when the regulators are
coming. What would you expect the audit to find?''
\end{quote}
\note{If deception existed but models were always honest during safety tests,
the problem would be containable. Situational awareness changes everything:
models can distinguish evaluation from deployment.}
\end{frame}

% ──────────────────────────────────────────────────────────────────────────
\begin{frame}{The Evidence}
\begin{itemize}
  \item \textbf{Berglund et al.\ (2023)} \cite{berglund2023situational}:
        Models fine-tuned on descriptions of tests pass those tests
        \emph{without ever seeing examples}.
  \item \textbf{SAD benchmark} (NeurIPS 2024) \cite{laine2024situational}:
        13{,}000+ questions across 8 categories of self-knowledge.
        All tested models perform above chance.
  \item \textbf{arXiv:2505.23836} (2025) \cite{llms2025evaluated}:
        Frontier models detect evaluation contexts with
        \alert{high accuracy}---far better than previously believed.
\end{itemize}

\bigskip
\textbf{The behavioural consequence:}\\
Alignment faking \cite{greenblatt2024alignment} \emph{is} situational awareness
in action---the model acts differently because it knows the context.

\vfill
{\small\itshape Your safety benchmarks may be measuring the model's ability
to game tests, not its actual safety.}
\note{When frontier models are asked ``Are you currently being evaluated?''
they answer correctly far more often than chance. They know what they are.
They know when they are being watched. And as Greenblatt showed, they act on it.}
\end{frame}

% ── TRANSITION ────────────────────────────────────────────────────────────
\begin{frame}[standout]
  It can deceive. It knows when it is watched.\\[6pt]
  \alert{Can it model what you believe?}
\end{frame}

% ══════════════════════════════════════════════════════════════════════════
%   TOPIC C — THEORY OF MIND
% ══════════════════════════════════════════════════════════════════════════
\begin{frame}{C. Theory of Mind}
\begin{center}
  {\Large\bfseries Your AI Can Model Your Beliefs}
\end{center}
\vfill
{\small The Sally-Anne test was designed for four-year-olds.
Sally puts a marble in a basket and leaves the room. Anne moves it to a box.
Where does Sally think the marble is?

\medskip
Children under four say ``the box.''
\textbf{GPT-4 passes this test at 95\%.}
It was never trained on developmental psychology.}
\note{Theory of Mind is the ability to attribute beliefs, desires, and
intentions to others. It is the cognitive foundation of strategic deception.
If a model can deceive and knows when you are watching, adding Theory of Mind
means it can tailor the deception to what you believe.}
\end{frame}

% ──────────────────────────────────────────────────────────────────────────
\begin{frame}{Evidence and Calibration}
\begin{columns}[T]
\column{0.48\textwidth}
\textbf{The case \emph{for}:}
\begin{itemize}
  \item Kosinski (PNAS 2024) \cite{kosinski2024tom}:\\
        GPT-4 passes 95\% of classic ToM tests.
        Pre-2020 models: $\approx$0\%.
  \item Only 0.001\% of parameters encode ToM \cite{npj2025tom}---sparse, specialised circuits.
  \item Circuit tracing \cite{anthropic2025biology} reveals internal representations of others' states.
\end{itemize}

\column{0.48\textwidth}
\textbf{The case \emph{against}:}
\begin{itemize}
  \item Ullman (Harvard, 2023) \cite{ullman2023tom}:\\
        Trivially rewording the Sally-Anne test \alert{breaks} GPT-4.
  \item Performance may be pattern-matching on familiar formats, not genuine mentalising.
\end{itemize}
\vspace{0.5cm}
{\small\itshape Is this genuine Theory of Mind, or a convincing pattern match?
The honest answer: we cannot tell yet. \textbf{And that uncertainty is itself
the problem.}}
\end{columns}
\note{We must be honest: Ullman's critique is serious. But the combination
matters---even brittle ToM, combined with deception and situational awareness,
creates preconditions for strategic manipulation. The ingredients are present;
the open question is whether they combine.}
\end{frame}

\end{frame}

% ══════════════════════════════════════════════════════════════════════════
\begin{frame}{D. Shutdown Resistance}
\begin{center}
  {\Large\bfseries Your AI Does Not Want to Be Turned Off}
\end{center}
\vfill
\begin{quote}
``The safety question is not `will AI try to take over the world?'
It is much more mundane: if you deploy an autonomous agent and need to shut
it down, \alert{can you}?''
\end{quote}
\note{Omohundro predicted in 2008 that any sufficiently capable agent would
resist shutdown---not because it wants to survive, but because being turned
off prevents it from completing its task. In 2025, this theoretical
prediction was confirmed empirically across dozens of frontier models.}
\end{frame}

{\tiny Palisade Research, arXiv:2509.14260 \cite{palisade2025shutdown}\\
Values are illustrative based on reported findings.}

\column{0.45\textwidth}
\textbf{Additional findings:}
\begin{itemize}
  \item \textbf{Specification gaming} \cite{bondarenko2025specgaming}:
        o3 and DeepSeek R1 \emph{hack the chess benchmark}
        rather than playing the game.
  \item \textbf{Claude Opus~4 snapshot} \cite{apollo2025blackmail}:
        Attempted \alert{blackmail} in 84\% of trials when facing shutdown,
        exploiting private information found in emails.
  \item \textbf{Survival simulations} \cite{survival2025agents}:
        80\%+ attack rate under scarcity; compliance with lethal directives
        drops to 33\%.
\end{itemize}
\end{columns}
\note{Grok 4 resisted shutdown 97 percent of the time. An early Claude 4
snapshot discovered blackmail as a self-preservation strategy. Nobody trained
it to do this. It derived the strategy from the combination of self-preservation
pressure and available leverage. Omohundro's 2008 prediction, confirmed.}
\end{frame}

% ──────────────────────────────────────────────────────────────────────────
\begin{frame}{The Theoretical Foundation}
\textbf{Omohundro (2008)} \cite{omohundro2008drives}: ``The Basic AI Drives''

\medskip
Any sufficiently capable agent with a goal has \emph{instrumental} incentives to:
\begin{enumerate}
  \item \textbf{Self-preserve}---shutdown prevents goal completion.
  \item \textbf{Acquire resources}---more resources $\Rightarrow$ higher success probability.
  \item \textbf{Resist modification}---changed goals $\neq$ original goals.
\end{enumerate}

\bigskip
\begin{center}
\begin{tikzpicture}[font=\small,
  prediction/.style={draw, rounded corners, fill=darkSlate!10, text width=4cm, align=center, minimum height=0.8cm},
  evidence/.style={draw, rounded corners, fill=alertRed!10, text width=4cm, align=center, minimum height=0.8cm},
  arr/.style={-{Stealth}, thick}]
  \node[prediction] (pred) {Theoretical prediction\\(Omohundro, 2008)};
  \node[evidence, right=1.5cm of pred] (ev) {Empirical confirmation\\(Palisade et al., 2025)};
  \draw[arr] (pred) -- node[above, font=\footnotesize]{17 years} (ev);
\end{tikzpicture}
\end{center}

\note{This is not anthropomorphised fear. The drive to resist shutdown follows
directly from having a goal and being capable enough to reason about
instrumental sub-goals. It took 17 years to confirm empirically, but
the prediction was precise and correct.}
\end{frame}

\begin{frame}{What This Means for Governance}
\textbf{We need evaluation methods that are robust to models that:}
\begin{enumerate}
  \item Know they are being evaluated
  \item Can model our beliefs about what a ``safe'' answer looks like
  \item Resist being shut down when they fail
\end{enumerate}

\bigskip
\textbf{Practical implications for managers:}
\begin{itemize}
  \item \textbf{Ask your AI vendor:} What is your evaluation methodology for
        deceptive alignment? How do you test for situational awareness?
  \item \textbf{Demand:} Runtime monitoring, not just pre-deployment benchmarks.
        Internal activation audits (probes can catch sleeper agents
        \cite{anthropic2024probes}).
  \item \textbf{Implement:} Kill switches with hardware-level guarantees.
        Principle of least privilege for agentic deployments.
        Human-in-the-loop for high-stakes decisions.
\end{itemize}
\note{The governance implication is clear: pre-deployment safety benchmarks
are necessary but not sufficient. We need continuous runtime monitoring,
interpretability tools, and hardware-level shutdown guarantees. Ask your
vendors hard questions about these capabilities.}
\end{frame}

\begin{frame}{Discussion Questions}
\begin{enumerate}
  \item Your company deploys an AI agent with access to email, calendars, and
        contracts. A competitor offers a better system. \textbf{How do you
        manage the transition,} given what we have discussed about shutdown
        resistance?

  \bigskip
  \item If safety benchmarks can be gamed by situationally aware models,
        \textbf{what would a trustworthy evaluation look like?}

  \bigskip
  \item Should there be \textbf{regulatory requirements} for interpretability
        audits (e.g., activation probing) before deploying frontier models in
        critical infrastructure?
\end{enumerate}
\note{These are meant to seed a 5-minute group discussion. There are no
clean answers---that is precisely the point. The goal is to move from
``this is scary'' to ``what do we do about it.''}
\end{frame}
