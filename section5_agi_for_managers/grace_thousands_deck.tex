\section{Moment 1 --- The thirteen-year shift}
% =========================================================================

\begin{frame}{The question}
  \vfill
  \centering
  {\Large \emph{``When unaided machines can accomplish}}\\[0.3em]
  {\Large \emph{every task better and more cheaply}}\\[0.3em]
  {\Large \emph{than human workers.''}}\\[1.5em]
  {\color{mute}They call it {\bfseries High-Level Machine Intelligence} (HLMI).}\\
  {\color{mute}Call it AGI if you prefer.}\\[0.8em]
  {\scriptsize\color{mute}Verbatim definition given to respondents, Grace et al.~(2024), p.~4.}
  \vfill
\end{frame}

\begin{frame}{In 2022, the aggregate answer was:}
  \vfill
  \centering
  \bignum{2060}\\[0.4em]
  {\large 50\% probability date for HLMI}\\[2em]
  {\color{mute}Roughly a generation away.}
  \vfill
\end{frame}

\begin{frame}{One year later, the same experts said:}
  \vfill
  \centering
  \bignum{2047}\\[0.4em]
  {\large 50\% probability date for HLMI}\\[2em]
  {\color{mute}Thirteen years closer.}
  \vfill
\end{frame}

\begin{frame}{The shift, as a picture}
  \centering
  \includegraphics[width=0.62\textwidth]{paper_assets/figures/fig3a_hlmi.png}\\[0.3em]
  {\small\color{mute}Aggregate forecast: 10\%, 50\%, and 90\% probability dates for HLMI, 2022 vs.~2023.}
\end{frame}

\begin{frame}{And for ``all human jobs''}
  \centering
  \includegraphics[width=0.62\textwidth]{paper_assets/figures/fig3b_faol.png}\\[0.3em]
  {\small\color{mute}Full Automation of Labor (FAOL) also shifted --- by 48 years.}

\begin{frame}{Read that again}
  \vfill
  \centering
  {\Large The 50\% date for automating}\\[0.3em]
  {\Large \textbf{\emph{every job on earth}}}\\[0.3em]
  {\Large moved from {\color{accent}2164} to {\color{accent}2116}.}
  \vfill
\end{frame}

\section{Moment 3 --- A tail that will not go away}
% =========================================================================

\begin{frame}{They were also asked about outcomes.}
  \vfill
  \centering
  {\large The exact question asked:}\\[0.8em]
  {\Large \emph{``Assume for the purpose of this question that}}\\[0.2em]
  {\Large \emph{HLMI will at some point exist. How positive}}\\[0.2em]
  {\Large \emph{or negative do you expect the overall impact}}\\[0.2em]
  {\Large \emph{of this to be on humanity, in the long run?''}}\\[1.2em]
  {\color{mute}Respondents assigned probabilities (summing to 100\%) across five buckets:}\\[0.3em]
  {\small extremely good $\cdot$ on balance good $\cdot$ neutral $\cdot$ on balance bad $\cdot$ extremely bad (e.g.\ human extinction)}\\[0.6em]
  {\scriptsize\color{mute}Verbatim, Grace et al.~(2024), footnote 5, p.~14; see also \S4.2, p.~12.}
  \vfill
\end{frame}

\section{So what?}
% =========================================================================

\begin{frame}{Three things to leave this room with}
  \vfill
  \begin{enumerate}\setlength\itemsep{1em}
    \item Expert timelines are \textbf{unstable}.\\
          {\small Do not anchor a five-year plan on a point estimate.}
    \item Narrow-task forecasts \textbf{already under-shot}.\\
          {\small Assume your workflow-level dot arrives early, not late.}
    \item The tail is \textbf{not a fringe}.\\
          {\small 38\% of the field assigns $\ge$10\% to catastrophic outcomes.}
  \end{enumerate}
  \vfill
\end{frame}

\begin{frame}{The one number}
  \vfill
  \centering
  \bignum{13}\\[0.4em]
  {\large years of expected progress}\\[0.3em]
  {\large compressed into \emph{twelve months} of calendar time.}\\[2em]
  {\color{mute}Plan as if that could happen again.}
  \vfill
\end{frame}

\begin{frame}[allowframebreaks]{Every quoted phrase, with page number}
  \footnotesize
  All quotations in this deck, verified against \emph{Grace et al.~(2024), ``Thousands of AI Authors on the Future of AI''}, arXiv:2401.02843.\\[0.8em]

  \begin{itemize}\setlength\itemsep{0.5em}
    \item \textbf{HLMI definition} --- ``unaided machines can accomplish every task better and more cheaply than human workers.''\\
      \emph{Source:} p.~4, verbatim from the survey instrument.

    \item \textbf{Outcomes question} --- ``Assume for the purpose of this question that HLMI will at some point exist. How positive or negative do you expect the overall impact of this to be on humanity, in the long run?''\\
      \emph{Source:} footnote~5, p.~14; summarised in \S4.2, p.~12.

    \item \textbf{Five outcome buckets} --- ``extremely good \dots\ on balance good \dots\ more or less neutral \dots\ on balance bad \dots\ extremely bad (e.g.\ human extinction).''\\
      \emph{Source:} footnote~5, p.~14.

    \item \textbf{``As bad as human extinction''} --- ``Between 37.8\% and 51.4\% of respondents gave at least a 10\% chance to advanced AI leading to outcomes as bad as human extinction.''\\
      \emph{Source:} abstract, p.~1; figure and discussion in \S4.2, p.~12.

    \item \textbf{HLMI aggregate dates} --- 2060 (2022 survey) $\to$ 2047 (2023 survey); 13-year shift.\\
      \emph{Source:} \S4.1, Figure~3a.

    \item \textbf{FAOL aggregate dates} --- 2164 (2022) $\to$ 2116 (2023); 48-year shift.\\
      \emph{Source:} \S4.1, Figure~3b.
  \end{itemize}
  \vspace{0.5em}
  \textbf{Not from the paper:} the ``bridge engineer / 10\% collapse'' analogy is the speaker's illustration of how to read a fat tail; it does not appear in Grace et al.
